🔒 Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Admins koleksiyonu - yalnız özü oxuya bilər
    match /admins/{uid} {
      allow read: if request.auth != null && request.auth.uid == uid;
      allow write: if isAdmin();
    }

    // API Açarları - yalnız adminlər
    match /api_keys/{doc} {
      allow read, write: if isAdmin();
    }

    // Parametrlər - yalnız adminlər yaza bilər, hamı oxuya
    match /settings/{doc} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // AI logları - hamı yaza bilər (sorğular üçün), admin oxuya
    match /ai_logs/{doc} {
      allow create: if true;
      allow read: if isAdmin();
    }

    // Admin logları - yalnız adminlər
    match /admin_logs/{doc} {
      allow read, write: if isAdmin();
    }

    function isAdmin() {
      return request.auth != null &&
             exists(/databases/$(database)/documents/admins/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.role == 'admin';
    }
  }
}
